
# Client API
 - get_commands () 'api/commands/list' (get request)
 - get_command (command_name: int) 'api/commands/get/<command_name:int>' (get request)
 - get_programs () 'api/programs/list' (get request)
 - get_status () 'api/status' (get request)
 - get_program_status (program_name: int) 'api/programs/status' (receives program_name in json body)

 - execute_command(command_name: int, API_key: str) 'api/commands/execute' (receives command_name and API_key in json body)
 - execute_program(program_name: int, API_key: str) 'api/programs/execute' (receives program_name and API_key in json body)
 - stop_program(program_name: int, API_key: str) 'api/programs/stop' (receives program_id and API_key in json body)
